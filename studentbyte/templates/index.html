📖 SQL Syntax Cheatsheet
CREATE TABLE students ( name VARCHAR(100), roll VARCHAR(50), phone VARCHAR(20), email VARCHAR(100), course VARCHAR(50), department VARCHAR(50) );
ALTER TABLE students ADD COLUMN year 1;
ALTER TABLE students DROP COLUMN year;
ALTER TABLE students RENAME TO alumni;
-- ⚠️ Deletes entire table! DROP TABLE students;
-- ⚠️ Removes all rows, keeps table TRUNCATE TABLE students;
RENAME TABLE students TO alumni;
COMMENT ON TABLE students IS 'Stores student records';
SELECT * FROM students;
SELECT * FROM students WHERE roll='CS001';
SELECT * FROM students LIMIT 5;
INSERT INTO students (name, roll, phone, email, course, department) VALUES ('Arjun', 'CS001', '9876543210', 'a@b.com', 'B.Tech', 'CSE');
-- Update one field UPDATE students SET name='Arjun' WHERE roll='CS001'; -- Update multiple fields UPDATE students SET name='Arjun', phone='9999999999' WHERE roll='CS001'; -- Update all records UPDATE students SET course='B.Tech';
-- Delete one student DELETE FROM students WHERE roll='CS001'; -- Delete all students DELETE FROM students;
-- Insert if not exists, update if exists (by roll) MERGE INTO students (name, roll, phone) VALUES ('Arjun', 'CS001', '9876543210');
-- Grant read access GRANT READ ON students TO user1; -- Grant all access GRANT ALL ON students TO user1;
-- Revoke read access REVOKE READ ON students FROM user1; -- Revoke all access REVOKE ALL ON students FROM user1;
